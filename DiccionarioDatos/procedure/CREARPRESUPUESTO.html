PROCEDURE CrearPresupuesto (
    p_ID_Presupuesto INT,
    p_Nombre VARCHAR2,
    p_Monto DECIMAL,
    p_FechaFin DATE
)
IS
BEGIN
    INSERT INTO Presupuesto (ID_Presupuesto, Nombre, Monto, FechaFin)
    VALUES (p_ID_Presupuesto, p_Nombre, p_Monto, p_FechaFin);

    INSERT INTO Modificacion (ID_Modificacion, ID_Presupuesto, TipoModificacion)
    VALUES (Modificacion_SEQ.NEXTVAL, p_ID_Presupuesto, 'Creación');

    COMMIT;
END;

-- Procedimiento para actualizar un presupuesto existente
CREATE OR REPLACE PROCEDURE ActualizarPresupuesto (
    p_ID_Presupuesto INT,
    p_NuevoMonto DECIMAL,
    p_NuevaFechaFin DATE
)
IS
BEGIN
    UPDATE Presupuesto
    SET Monto = p_NuevoMonto, FechaFin = p_NuevaFechaFin
    WHERE ID_Presupuesto = p_ID_Presupuesto;

    INSERT INTO Modificacion (ID_Modificacion, ID_Presupuesto, TipoModificacion)
    VALUES (Modificacion_SEQ.NEXTVAL, p_ID_Presupuesto, 'Actualización');

    COMMIT;
END;
